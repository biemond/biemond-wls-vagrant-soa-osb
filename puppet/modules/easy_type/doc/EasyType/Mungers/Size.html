def unsafe_munge(size)
  case size
  when /^\d+(K|k)$/ then size.chop.to_i * 1024
  when /^\d+(M|m)$/ then size.chop.to_i * 1024 * 1024
  when /^\d+(G|g)$/ then size.chop.to_i * 1024 * 1024 * 1024
  when /^\d+$/ then size.to_i
  else
    fail("invalid size")
  end
end
